         double currentRadius = (i % 2 == 0) ? radius1 : radius2;

      yPoints[i] = (int) (centery - radius * Math.sin(angle));
   }
   
   // Draw the star using the Polygon object
   Polygon star = new Polygon(xPoints, yPoints, 5);
